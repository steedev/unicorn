### 3. Private

```js
class Person {
  #name
  #age
  constructor(name, age) {
    this.#name = name
    this.#age = age
  }

  sayHello() {
    return `${this.#name} ${this.#age}`
  }
}

const person = new Person('Kuba', 20)

person // {}
```
